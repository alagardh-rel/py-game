# Word Search Game - Python Dependencies
# ========================================
# These are all the Python packages needed to run the application

# Flask: Lightweight web framework for building web applications
# Provides routing, template rendering, and request handling
Flask==2.3.3

# Werkzeug: Utility library for WSGI (Web Server Gateway Interface)
# Required by Flask for request/response handling
Werkzeug==2.3.7

# Jinja2: Template engine used by Flask for rendering HTML
# Allows dynamic content generation in templates
Jinja2==3.1.2

# MarkupSafe: String escaping utilities for safe HTML rendering
# Used by Jinja2 for security
MarkupSafe==2.1.3

# Click: Command-line interface creation kit
# Required by Flask for command-line utilities
Click==8.1.7

# itsdangerous: Data signing and timestamp signing library
# Used by Flask for session security
itsdangerous==2.1.2

# Installation Instructions:
# =========================
# 1. Make sure you have Python 3.7+ installed
# 2. Create a virtual environment (optional but recommended):
#    python -m venv venv
# 3. Activate the virtual environment:
#    - Windows: venv\Scripts\activate
#    - Mac/Linux: source venv/bin/activate
# 4. Install dependencies:
#    pip install -r requirements.txt
# 5. Run the app:
#    python app.py
# 6. Open your browser to: http://localhost:5000
